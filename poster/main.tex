\begin{enumerate}
    \item Connect to the \link{\thewifi} WiFi;
    \item Go to \link{\theintranet} or flash this QR Code :
\begin{center}
\qrcode[height=0.3\linewidth]{\theintranet}
\end{center}
    \item If you don't have an account yet:
        \begin{enumerate}
            \item Click on 'Sign In';
            \item Do what the website tells you to do.
        \end{enumerate}
    \item If you have an account, click on 'Log In', then log in;
    \item Go to you user page; 
        \begin{itemize}
            \item {\centering%
\includegraphics[width=.55\linewidth]{pane_ok.png}}

            Your account is active!


            \item {\centering%
\includegraphics[width=.55\linewidth]{pane_nok.png}}

            Your account is not active. Click on 'Pay a connection'.
        \end{itemize}
    \item Pay a connection (if you need it)
    \begin{enumerate}
        \item {Select the payment method} : 

        {\centering%
\includegraphics[width=.55\linewidth]{pay_method.png}}

        \item Select your articles. As an example if I stay 3 months, I select 'Abonnement 1 mois' and display '3' in the spin box.

        {\centering%
\includegraphics[width=.55\linewidth]{pay_article.png}}

        \item Click on confirm and follow the website.
    \end{enumerate}
    \item You can use the WiFi!

Go to \link{\thewebsite} if you don't know how to connect.

\end{enumerate}
